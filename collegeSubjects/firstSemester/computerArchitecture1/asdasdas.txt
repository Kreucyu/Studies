	//VOU UTILIZAR A POSICAO DE MEMORIA #FF PARA SINALIZAR SE HOUVE TROCA OU NAO
	//01-->#FF (HOUVE TROCA) 00-->#FF (NAO HOUVE TROCA)
	
	POP D;
	POP E;
	MOV D, A;
	MOV A, #B;
	MOV E, A;
	MOV A, #C;
	MOV B, A;
	ADD 01, B;
	MOV C, A;
	ADD 01, C;
	//COMPARA SE C ==> 80
	MOV C, A;
	SUB 80, A;
	JMPZ FECHOUCICLO;
	//ELSE
	JMP COMPARA;

FECHOUCICLO:
	MOV #FF, A;
	SUB 00, A; //COMPARA SE ACHOU 00
	JMPZ ACABOUORDENACAO;
	JMP INICIOCICLO;

ACABOUORDENACAO:
	JMP ACABOUORDENACAO;
